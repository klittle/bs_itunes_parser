    def parse(itunes_xml_file_name)

      @doc = Nokogiri::XML(itunes_xml_file_name)

      #version_key is a Nokogiri::XML::Element
      version_key = @doc.xpath('/plist/dict/string[1]')[0]
      self.version = version_key.content

      self.parse_songs
      self.parse_playlists            
    end
